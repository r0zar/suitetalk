# Phase 2: Feed — Implementation Plan

> **Spec reference:** [`docs/mvp-spec.md`](../mvp-spec.md), §6 "Data model" and §11 "Demo flow".
>
> **Process note:** Same as Phase 1 — verifiable runtime checkpoints over TDD. Testing waived for MVP.

**Goal:** The chat screen becomes a real-time shared feed. Notes are Firestore docs read via `onSnapshot`. The main feed is read-only on the client; the only writers in MVP are server-side (Phase 5+) and a temporary debug-page button we add now for verification.

**Architecture:** A single `notes` Firestore collection. Each client subscribes via `onSnapshot` ordered by `createdAt desc`, limited to the last N. Rendering uses the existing themed `ScrollView` + `messageBubble` pattern. The AI chat (`useChat`) and the `/api/chat` route are removed. A "Post test note" button is added to the existing Debug page so we can verify the realtime path before the voice pipeline lands.

**Tech stack:** `@react-native-firebase/firestore`, existing themed components, `expo-router`.

## Data shape

`notes/{noteId}`:

```ts
{
  authorUid: string;
  authorHandle: string;   // denormalized at write time
  text: string;
  createdAt: Timestamp;   // server timestamp
}
```

Order: `createdAt desc`. Limit: `100` (newest first; we'll render reversed so latest sits at the bottom like a normal chat).

## Task 1: Notes service module

**Files:**
- Create: `src/lib/notes.ts`

- [ ] **Step 1: Create the module**

  ```ts
  import firestore, {
    FirebaseFirestoreTypes,
  } from '@react-native-firebase/firestore';

  export type Note = {
    id: string;
    authorUid: string;
    authorHandle: string;
    text: string;
    createdAt: Date | null; // null while serverTimestamp is pending
  };

  const NOTES_LIMIT = 100;

  function notesQuery() {
    return firestore()
      .collection('notes')
      .orderBy('createdAt', 'desc')
      .limit(NOTES_LIMIT);
  }

  function toNote(
    snap: FirebaseFirestoreTypes.QueryDocumentSnapshot,
  ): Note {
    const data = snap.data();
    const ts = data.createdAt as FirebaseFirestoreTypes.Timestamp | null;
    return {
      id: snap.id,
      authorUid: data.authorUid ?? '',
      authorHandle: data.authorHandle ?? '',
      text: data.text ?? '',
      createdAt: ts ? ts.toDate() : null,
    };
  }

  export function subscribeToNotes(
    onChange: (notes: Note[]) => void,
    onError?: (err: Error) => void,
  ): () => void {
    return notesQuery().onSnapshot(
      (qs) => onChange(qs.docs.map(toNote)),
      (err) => onError?.(err),
    );
  }

  export async function postNote(input: {
    text: string;
    authorUid: string;
    authorHandle: string;
  }): Promise<string> {
    const trimmed = input.text.trim();
    if (!trimmed) throw new Error('Note text cannot be empty.');
    const ref = await firestore().collection('notes').add({
      text: trimmed,
      authorUid: input.authorUid,
      authorHandle: input.authorHandle,
      createdAt: firestore.FieldValue.serverTimestamp(),
    });
    return ref.id;
  }
  ```

## Task 2: `useNotes` hook

**Files:**
- Create: `src/hooks/use-notes.ts`

- [ ] **Step 1: Create the hook**

  ```ts
  import { useEffect, useState } from 'react';

  import { subscribeToNotes, type Note } from '@/lib/notes';

  type State =
    | { status: 'loading'; notes: Note[] }
    | { status: 'ready'; notes: Note[] }
    | { status: 'error'; notes: Note[]; error: string };

  export function useNotes() {
    const [state, setState] = useState<State>({ status: 'loading', notes: [] });

    useEffect(() => {
      const unsub = subscribeToNotes(
        (notes) => setState({ status: 'ready', notes }),
        (err) =>
          setState((s) => ({
            status: 'error',
            notes: s.notes,
            error: err.message,
          })),
      );
      return () => unsub();
    }, []);

    return state;
  }
  ```

## Task 6: Update Firestore rules to allow authenticated note writes

**Files:**
- Modify: `firestore.rules`

The current rules deny all writes to `notes/`. The debug "Post test note" button is a client write, so we need to allow authenticated creates (still no updates/deletes for MVP — notes are append-only).

- [ ] **Step 1: Update the file**

  Replace the entire contents of `firestore.rules` with:

  ```
  rules_version = '2';
  service cloud.firestore {
    match /databases/{database}/documents {
      // Users may read any handle (needed for displaying authors), and only
      // write their own user doc.
      match /users/{uid} {
        allow read: if request.auth != null;
        allow create, update: if request.auth != null && request.auth.uid == uid;
      }

      // Notes: authenticated users can read and append. No updates or deletes.
      // The authorUid on a new note must match the writer's auth UID; the
      // server-side path (Phase 5) will use Admin SDK and bypass these rules.
      match /notes/{noteId} {
        allow read: if request.auth != null;
        allow create: if
          request.auth != null
          && request.resource.data.authorUid == request.auth.uid
          && request.resource.data.text is string
          && request.resource.data.text.size() > 0
          && request.resource.data.text.size() <= 2000
          && request.resource.data.authorHandle is string;
        allow update, delete: if false;
      }
    }
  }
  ```
